    </audio>

    <button id="play-toggle">▶️ Play</button>

    <div id="now-playing">Loading current track...</div>

    <!-- Artwork will be inserted here -->

  </div>


  <script>

    const audio = document.getElementById('stream');

    const playBtn = document.getElementById('play-toggle');

    const nowPlaying = document.getElementById('now-playing');

    let isPlaying = false;


    playBtn.addEventListener('click', () => {

      if (!isPlaying) {

        audio.play().catch(err => console.error("Play failed:", err));

        playBtn.textContent = '⏸ Pause';

        isPlaying = true;

      } else {

        audio.pause();

        playBtn.textContent = '▶️ Play';

        isPlaying = false;

      }

    });


    async function fetchNowPlaying() {

      try {

        const res = await fetch('/api/metadata');

        const data = await res.json();


        nowPlaying.textContent = `Now Playing: ${data.nowPlaying || "Unknown Track"}`;


        // Show artwork if present

        let artBox = document.getElementById('artwork');

        if (data.artwork) {

          if (!artBox) {

            artBox = document.createElement('img');

            artBox.id = 'artwork';

            document.getElementById('player-box').appendChild(artBox);

          }

          artBox.src = data.artwork;

          artBox.alt = "Artwork";

        } else if (artBox) {

          artBox.remove();

        }

      } catch (err) {

        nowPlaying.textContent = "Unable to fetch metadata";

        console.error("Fetch error:", err);

      }

    }


    setInterval(fetchNowPlaying, 15000);

    fetchNowPlaying();

  </script>
